# 字符串转数字

# 1、纯数字转换

var s = "234";

    1. 字符串在运算操作中会被当做数字类型来处理
      s * 1 // 234
    1. 字符前加“+”
      +s // 234
    1. string的两个转换函数，只对string有效
      parseInt(s); // 234
      parseFloat(s); // 234
    1. 强制类型转换
      Number(s); // 234

# 2、数字加字母等非数字转换

var s = '234string';
parseInt(s);　// 234
parseFloat(s); // 234.0
